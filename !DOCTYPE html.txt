    // Select Payment Mode - FIXED VERSION with debouncing
    function selectPaymentMode(mode) {
      // Prevent duplicate calls within 500ms
      if (mode === lastPaymentMode && paymentModeTimeout) {
        return;
      }
      
      lastPaymentMode = mode;
      selectedPaymentMode = mode;
      
      // Clear any existing timeout
      if (paymentModeTimeout) {
        clearTimeout(paymentModeTimeout);
      }
      
      // Update radio buttons
      document.querySelectorAll('input[name="paymentMode"]').forEach(radio => {
        radio.checked = radio.value === mode;
      });
      
      // Update label styling
      document.querySelectorAll('.payment-option').forEach(option => {
        option.classList.remove('selected');
      });
      
      const selectedOption = document.querySelector(`.payment-option input[value="${mode}"]`);
      if (selectedOption) {
        selectedOption.parentElement.classList.add('selected');
      }
      
      // Voice feedback - with debouncing
      if (voiceAssistantEnabled) {
        // Debounce the speech to prevent duplicates
        paymentModeTimeout = setTimeout(() => {
          speak(`Payment mode set to ${mode}`);
          paymentModeTimeout = null;
        }, 100);
      }
    }

    function speak(text) {

    }
    
    function addToCart(name, price, isVoiceOrder = false, skipVoiceFeedback = false) {
      // Check if item already in cart
      const existingItem = cart.find(item => item.name === name);
      
      if (existingItem) {
        // If item exists, increase quantity instead of price
        existingItem.quantity++;
        existingItem.totalPrice = existingItem.price * existingItem.quantity;
        total += price;
        
        if (isVoiceOrder && !skipVoiceFeedback) {
          // DON'T call speak() here - let the caller handle it
        } else if (!skipVoiceFeedback) {
          showOrderConfirmation(`Another ${name} added`, price);
        }
      } else {
        // Add new item to cart
        cart.push({ 
          name: name, 
          price: price,
          quantity: 1,
          totalPrice: price
        });
        total += price;
        
        if (isVoiceOrder && !skipVoiceFeedback) {
          // DON'T call speak() here - let the caller handle it
        } else if (!skipVoiceFeedback) {
          showOrderConfirmation(name, price);
        }
      }
      
      updateCart();
    }
    
    function changeQuantity(index, change, skipVoiceFeedback = false) {
      const item = cart[index];
      const newQuantity = item.quantity + change;
      
      if (newQuantity < 1) {
        // If quantity would be 0, remove the item
        removeFromCart(index, skipVoiceFeedback);
        return;
      }
      
      // Update quantity
      item.quantity = newQuantity;
      const oldTotalPrice = item.totalPrice;
      item.totalPrice = item.price * newQuantity;
      
      // Update total
      total = total - oldTotalPrice + item.totalPrice;
      
      updateCart();
      if (!skipVoiceFeedback) {
        showOrderConfirmation(`${item.name} quantity: ${newQuantity}`, item.totalPrice);
      }
    }
    
    function removeFromCart(index, skipVoiceFeedback = false) {
      const item = cart[index];
      total -= item.totalPrice;
      
      cart.splice(index, 1);
      updateCart();
      if (!skipVoiceFeedback) {
        showOrderConfirmation(`${item.name} removed`, item.totalPrice);
      }
    }

    async function checkout() {
      const tableNo = document.getElementById('tableNo').value;
      if (!tableNo) {
        alert('Please enter a table number');
        return;
      }
      
      if (cart.length === 0) {
        alert("Your cart is empty!");
        return;
      }
      
      currentTableNo = tableNo;
      
      // Save cart items for bill generation BEFORE clearing
      const cartItemsForBill = JSON.parse(JSON.stringify(cart));
      const totalForBill = total;
      
      try {
        const response = await fetch('http://localhost:4000/api/orders', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify({
            userName: localStorage.getItem("userName") || "Guest",
            tableNo: tableNo,
            items: cartItemsForBill, // Use saved cart items
            total: totalForBill,
            paymentMode: selectedPaymentMode
          })
        });
        
        const result = await response.json();
        
        if (response.ok) {
          currentBillNo = result.order.billNo;
          generateBill(tableNo, result.order.billNo, cartItemsForBill, totalForBill, true);
        } else {
          alert('Error saving order: ' + result.error);
        }
      } catch (error) {
        console.log("Server not available, using localStorage fallback");
        currentBillNo = 'SB' + Date.now();
        generateBill(tableNo, currentBillNo, cartItemsForBill, totalForBill, false);
      }
    }

      let paymentClass = '';
      
      switch(selectedPaymentMode) {
        case 'Cash':
          paymentIcon = '💵';
          paymentClass = 'payment-cash';
          break;
        case 'UPI':
          paymentIcon = '📱';
          paymentClass = 'payment-upi';
          break;
        case 'Card':
          paymentIcon = '💳';
          paymentClass = 'payment-card';
          break;
      }
      
      billPaymentElement.className = `bill-payment-display ${paymentClass}`;
      billPaymentElement.innerHTML = `${paymentIcon} ${selectedPaymentMode}`;
      
      // Update bill items - use the saved cart items
      const billItems = document.getElementById('bill-items');
      billItems.innerHTML = '';
      
      // Check if cartItemsForBill is defined and has items
      if (cartItemsForBill && cartItemsForBill.length > 0) {
        cartItemsForBill.forEach(item => {
          const li = document.createElement('li');
          li.textContent = `${item.name} (${item.quantity} × ₹${item.price}) = ₹${item.totalPrice}`;
          li.style.padding = '10px 0';
          li.style.borderBottom = '1px solid #eee';
          billItems.appendChild(li);
        });
      } else {
        // Fallback: use current cart if available
        if (cart && cart.length > 0) {
          cart.forEach(item => {
            const li = document.createElement('li');
            li.textContent = `${item.name} (${item.quantity} × ₹${item.price}) = ₹${item.totalPrice}`;
            li.style.padding = '10px 0';
            li.style.borderBottom = '1px solid #eee';
            billItems.appendChild(li);
          });
        } else {
          // Show empty message
          const li = document.createElement('li');
          li.textContent = 'No items in order';
          li.style.padding = '10px 0';
          li.style.color = '#999';
          li.style.fontStyle = 'italic';
          billItems.appendChild(li);
        }
      }
      
      // Save order to localStorage
      let orders = JSON.parse(localStorage.getItem('orders')) || [];
      orders.push({
        userName: localStorage.getItem("userName") || "Guest",
        tableNo: tableNo,
        billNo: billNo,
        date: new Date().toLocaleString(),
        items: cartItemsForBill || cart, // Save the cart items
        total: totalForBill || total, // Save the total
        paymentMode: selectedPaymentMode,
        status: 'completed'
      });
      localStorage.setItem('orders', JSON.stringify(orders));
      
      // Show bill
      document.getElementById('bill').style.display = 'block';
      document.getElementById('bill').scrollIntoView({ behavior: 'smooth' });
      
      // Show success message
      showOrderConfirmation(`Order placed! Table: ${tableNo}`, totalForBill || total);
      
      // Clear cart only after bill is generated
      cart = [];
      total = 0;
      updateCart();
      document.getElementById('tableNo').value = '';
      
      // Voice confirmation
      if (voiceAssistantEnabled) {
        speak(`Order placed successfully for table ${tableNo}. Payment mode: ${selectedPaymentMode}`);
      }
    }

    // Send message to chatbot API
    async function sendChatbotMessage() {
      const message = chatbotInput.value.trim();
      if (!message) return;
      
      // Add user message to chat
      addChatbotMessage(message, 'user');
      chatbotInput.value = '';
      
      // Show typing indicator
      showTypingIndicator();
      
      try {
        // Call the chatbot API
        const response = await fetch('http://localhost:5000/chat', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify({
            message: message,
            cart: cart,
            total: total,
            tableNo: document.getElementById('tableNo').value || '',
            paymentMode: selectedPaymentMode
          })
        });
        
        const data = await response.json();
        
        // Remove typing indicator
        removeTypingIndicator();
        
        // Add bot response
        addChatbotMessage(data.response, 'bot');
        
        // Handle special actions
        if (data.action === 'add_to_cart') {
          const itemName = data.item;
          const itemPrice = data.price || 0;
          addToCart(itemName, itemPrice, false, true);
          showOrderConfirmation(itemName, itemPrice);
        }
        else if (data.action === 'remove_from_cart') {
          const itemIndex = cart.findIndex(item => item.name.toLowerCase() === data.item.toLowerCase());
          if (itemIndex !== -1) {
            removeFromCart(itemIndex, true);
            showOrderConfirmation(`${data.item} removed`, cart[itemIndex]?.price || 0);
          }
        }
        else if (data.action === 'checkout') {
          if (cart.length > 0) {
            checkout();
          }
        }
        
      } catch (error) {
        console.error('Chatbot API error:', error);
        removeTypingIndicator();
        
        // Fallback response
        addChatbotMessage("I'm having trouble connecting right now. Please try again or use the menu to place your order.", 'bot');
      }
    }

    // Voice order function
    async function startVoiceOrder() {
      const micButton = document.getElementById('micButton');
      
      if (isVoiceOrderActive) {
        addChatbotMessage("Please wait, I'm already processing a voice command.", 'bot');
        return;
      }
      
      isVoiceOrderActive = true;
      micButton.classList.add('active');
      micButton.disabled = true;
      
      addChatbotMessage("🎤 Listening... Please speak your order", 'bot');
      
      try {
        const response = await fetch('http://localhost:5000/voice-order');
        const data = await response.json();
        
        if (data.status === 'success') {
          // Process the voice command
          const command = data.command || 'add';
          const item = data.item;
          const quantity = data.quantity || 1;
          const itemData = voiceMenuItems[item.toLowerCase()];
          
          if (itemData) {
            if (command === 'add') {
              for (let i = 0; i < quantity; i++) {
                addToCart(itemData.name, itemData.price, true, true);
              }
              addChatbotMessage(`✅ Added ${quantity} ${itemData.name}(s) to your cart`, 'bot');
              speak(`Added ${quantity} ${itemData.name} to cart`);
            } else if (command === 'remove') {
              const itemIndex = cart.findIndex(cartItem => 
                cartItem.name.toLowerCase() === itemData.name.toLowerCase()
              );
              
              if (itemIndex !== -1) {
                const cartItem = cart[itemIndex];
                const currentQuantity = cartItem.quantity;
                
                if (currentQuantity > quantity) {
                  changeQuantity(itemIndex, -quantity, true);
                  addChatbotMessage(`➖ Removed ${quantity} ${itemData.name}(s) from cart`, 'bot');
                  speak(`Removed ${quantity} ${itemData.name} from cart`);
                } else {
                  removeFromCart(itemIndex, true);
                  addChatbotMessage(`🗑️ Removed ${itemData.name} from cart`, 'bot');
                  speak(`Removed ${itemData.name} from cart`);
                }
              } else {
                addChatbotMessage(`❌ ${itemData.name} not found in your cart`, 'bot');
                speak(`${itemData.name} not found in your cart`);
              }
            }
          } else {
            addChatbotMessage(`❌ I couldn't find "${item}" in our menu. Please try again.`, 'bot');
            speak(`Item ${item} not found in menu`);
          }
        } else {
          addChatbotMessage("❌ I couldn't understand that. Please try again.", 'bot');
          speak("I couldn't understand that. Please try again.");
        }
      } catch (error) {
        console.error('Voice order error:', error);
        addChatbotMessage("⚠️ Voice service is currently unavailable. Please type your order instead.", 'bot');
        speak("Voice service unavailable. Please type your order.");
      } finally {
        micButton.classList.remove('active');
        micButton.disabled = false;
        isVoiceOrderActive = false;
      }
    }
